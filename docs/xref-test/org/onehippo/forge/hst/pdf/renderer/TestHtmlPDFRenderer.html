36  public class TestHtmlPDFRenderer {
37  
38      private HtmlPDFRenderer pdfRenderer;
39  
40      @Before
41      public void setUp() throws Exception {
42          Properties tidyProps = new Properties();
43          InputStream input = getClass().getResourceAsStream("tidy.properties");
44          tidyProps.load(input);
45          input.close();
46          pdfRenderer = new HtmlPDFRenderer(tidyProps);
47          File cssFile = FileUtils.toFile(getClass().getResource("default-pdf-renderer.css"));
48          pdfRenderer.setCssURIs(new URI [] { cssFile.toURI() });
49      }
50  
51      @Test
52      public void testHtmlToPDF() throws Exception {
53          String htmlFileName = "solar-power-the-sky-is-the-limit.html";
54          InputStream htmlInput = null;
55          OutputStream pdfOutput = null;
56  
57          try {
58              htmlInput = getClass().getResourceAsStream(htmlFileName);
59              pdfOutput = new FileOutputStream(new File(new File("target"), htmlFileName + ".pdf"));
60              String documentURL = "http://localhost:8080/site/pdf/news/2011/solar-power-the-sky-is-the-limit.html";
61              String externalLinkBaseURL = "http://localhost:8080";
62              pdfRenderer.renderHtmlToPDF(htmlInput, "UTF-8", true, pdfOutput, documentURL, externalLinkBaseURL);
63          } finally {
64              IOUtils.closeQuietly(pdfOutput);
65              IOUtils.closeQuietly(htmlInput);
66          }
67      }
